{{/* buttons.html — PDF, arXiv, DOI, and BibTeX buttons for a publication. The BibTeX button carries the generated BibTeX in a data-bibtex attribute. Bootstrap's 'show.bs.modal' event handler in bibtex_modal.html reads data-bibtex from the triggering button and fills the modal's textarea. " and " }} {{ end }} {{ $bib := printf "@%s{%s,\n" $pubtype $citekey }} {{ $bib = printf "%s author = {%s},\n" $bib $authorStr }} {{ $bib = printf "%s title = {%s},\n" $bib $page.Title }} {{ $bib = printf "%s year = {%s},\n" $bib $year }} {{ with $page.Params.venue }}{{ $bib = printf "%s booktitle = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.booktitle }}{{ $bib = printf "%s booktitle = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.publisher }}{{ $bib = printf "%s publisher = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.volume }}{{ $bib = printf "%s volume = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.number }}{{ $bib = printf "%s number = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.pages }}{{ $bib = printf "%s pages = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.doi }}{{ $bib = printf "%s doi = {%s},\n" $bib . }}{{ end }} {{ with $page.Params.arxiv }}{{ $bib = printf "%s eprint = {%s},\n" $bib . }}{{ end }} {{ $bib = printf "%s}" $bib }} {{/* --- Button row --- */}}
